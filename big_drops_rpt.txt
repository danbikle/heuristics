DROP   TABLE IF EXISTS td08;
DROP TABLE
CREATE TABLE td08 AS
SELECT
tkr
,ydate
,AVG(closing_price)      cp
,AVG(closing_price_orig) cpo
FROM ydata
WHERE ydate + (365 *30) > (SELECT MAX(ydate) FROM ydata)
GROUP BY tkr,ydate
ORDER BY tkr,ydate
;

,cp
